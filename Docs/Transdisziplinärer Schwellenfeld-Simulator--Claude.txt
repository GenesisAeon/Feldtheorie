def universal_sigmoid(R, theta, beta, A, baseline=0):
    """
    Universelle Sigmoid-Funktion fÃ¼r Schwellendynamik
    
    Parameters:
    -----------
    R : array-like
        Stimulus (NektarqualitÃ¤t, ModellgrÃ¶ÃŸe, Akkretionsrate)
    theta : float
        Schwellenwert Î˜_R (Ãœbergangs-Mittelpunkt)
    beta : float
        Steilheit des Ãœbergangs (hÃ¶her = schÃ¤rfer)
    A : float
        Maximale Response-Amplitude
    baseline : float
        Response-Baseline (Offset)
    
    Returns:
    --------
    response : array-like
        Systemantwort (Tanzrate, Accuracy, QPO-Amplitude)
    """
    return baseline + A / (1 + np.exp(-beta * (R - theta)))

def fit_sigmoid_with_errors(x, y, y_err, domain_name):
    """
    Fittet Sigmoid mit Fehlerfortpflanzung
    
    Returns:
    --------
    params : array
        [theta, beta, A, baseline]
    cov : array
        Kovarianzmatrix
    """
    # Initiale SchÃ¤tzung
    theta_init = x[np.argmax(np.gradient(y))]
    A_init = np.max(y) - np.min(y)
    baseline_init = np.min(y)
    beta_init = 4.0
    
    p0 = [theta_init, beta_init, A_init, baseline_init]
    
    # Bounds (physikalisch sinnvoll)
    bounds = (
        [x.min(), 0.1, 0, 0],  # lower
        [x.max(), 20, A_init*2, baseline_init*2]  # upper
    )
    
    # Weighted fit
    try:
        params, cov = curve_fit(
            universal_sigmoid, x, y, 
            p0=p0, 
            sigma=y_err,
            absolute_sigma=True,
            bounds=bounds,
            maxfev=10000
        )
    except RuntimeError:
        print(f"âš ï¸  {domain_name}: Fit konvergiert nicht - verwende p0")
        params, cov = p0, np.eye(4) * np.inf
    
    return params, cov

def normalize_response(R, theta, beta, response):
    """Normalisiert Response auf [0,1] mittels Sigmoid"""
    R_norm = beta * (R - theta)
    response_norm = (response - response.min()) / (response.max() - response.min())
    return R_norm, response_norm
