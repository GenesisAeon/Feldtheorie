# CRITICAL BUGFIX: beta_band_mean was calculating CI width, not beta mean

import numpy as np
from dataclasses import dataclass
from typing import List

@dataclass
class TippingElement:
    """Represents a planetary tipping element with fitted parameters"""
    name: str
    beta: float
    theta: float
    steepness_band: tuple  # (lower, upper) confidence interval for beta
    
def analyze_tipping_elements(elements: List[TippingElement]) -> dict:
    """
    Corrected analysis of tipping elements.
    
    Args:
        elements: List of fitted tipping elements
        
    Returns:
        Dictionary with corrected evidence metrics
    """
    # Extract all beta values
    betas = np.array([e.beta for e in elements])
    
    # Calculate CI widths for transparency
    ci_widths = np.array([e.steepness_band[1] - e.steepness_band[0] 
                          for e in elements])
    
    # === CORRECTED METRICS ===
    beta_mean = np.mean(betas)
    beta_std = np.std(betas, ddof=1)  # Sample standard deviation
    beta_sem = beta_std / np.sqrt(len(betas))  # Standard error
    
    # CI width statistics (for uncertainty assessment)
    ci_width_mean = np.mean(ci_widths)
    ci_width_std = np.std(ci_widths, ddof=1)
    
    # === EVIDENCE PACKAGE ===
    evidence = {
        # PRIMARY METRICS (CORRECTED)
        "beta_mean": float(beta_mean),
        "beta_std": float(beta_std),
        "beta_sem": float(beta_sem),
        "beta_95ci": (float(beta_mean - 1.96*beta_sem), 
                      float(beta_mean + 1.96*beta_sem)),
        
        # SECONDARY METRICS (for transparency)
        "beta_band_width_mean": float(ci_width_mean),
        "beta_band_width_std": float(ci_width_std),
        
        # CONVERGENCE TEST
        "convergence_to_42": abs(beta_mean - 4.2),
        "within_tolerance": abs(beta_mean - 4.2) < 1.0,
        
        # RAW DATA (for reproducibility)
        "beta_values": betas.tolist(),
        "sample_size": len(betas)
    }
    
    return {
        "evidence": evidence,
        "elements": [
            {
                "name": e.name,
                "beta": e.beta,
                "theta": e.theta,
                "ci": e.steepness_band
            }
            for e in elements
        ]
    }
